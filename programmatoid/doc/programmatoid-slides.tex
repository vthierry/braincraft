\begin{frame}
  \frametitle{The notion of ``programmatoid computation'':}
- It is an input-output \hrefc{https://en.wikipedia.org/wiki/Straight-line_program}{straight-line program} implementing an operator on either \\\tab - binary values $\in \{0, 1\}$ \\\tab - or continuous normalized values $\in [0, 1]$, \\ which evolution is defined using an LN-system with linear units or the step-function $\mbox{H}(\cdot)$ as non-linearity:
\eqline{\begin{array}{rcl} o_n[t] &=& \left\{ \begin{array}{r} \mbox{H}(z_n[t]) \\ z_n[t] \\ \end{array} \right., \\ z_n[t] &\defq& \sum_{m\in\{1,M\}} w_{nm} \, i_m[t-1] + w_{n0}, n \in \{1, N\} \\ \end{array}}
where $i_m[t]$ is the $m$-th input at a discrete $t$, and $o_n[t]$ an output.
- A generalized programmatoid computation includes exponential and logarithm as non-linearity, when not other \hrefc{https://en.wikipedia.org/wiki/Pure_function} pure function.
\end{frame}

\begin{frame}
  \frametitle{The notion of ``sugared straight-line program'':}
  - It is an input-output piece of code with \\ \tab - instruction sequences, assigments, conditional instructions, and iterations (no general loops); \\ \tab - using [generalized] programmatoid computation in assignment. \\~\\ \begin{itemize}
  \item This implements the set of \hrefc{https://en.wikipedia.org/wiki/Primitive_recursive_function}{primitive recursive functions}.
  \item Any primitive recursive functions rewrites as a programmatoid.
\end{itemize}
~\\ - The key idea: \\\tab programmatoid implements conditional expressions \\ on binary or numerical values, i.e.: \centerline{\tt \tiny value = if condition then true\_value else false\_value} rewrites as \centerline{\tt \tiny value = H(condition) true\_value + H(not condition) false\_value} 
\end{frame}

  \frametitle{The reasons why (not a formal proof):}
\centerline{\begin{tabular}[t]{cc}\hphantom{1mm}\parbox[t]{0.4\textwidth}{~\\\begin{itemize}
      \item Iterations unfolding \\~\\~\\ ~\\~\\
      \item Conditional  expansion  \\~\\~\\
      \item Conditional reduction
    \end{itemize}} &\parbox[t]{0.6\textwidth}{\tiny
      \rule{0.5\textwidth}{0.4pt} 
      \centerline{\parbox{0.5\textwidth}{\tt for n from 0 to n\_1 ≤ n\_max do \\\tab instructions\_1  endfor}}
      ~\\{$\longrightarrow$}\\~
      \centerline{\parbox{0.5\textwidth}{\tt  if n ≥ 0 then \\\tab substitute(n =  0, instructions\_1)\\ ...\\if n = n\_1 then \\\tab substitute(n =  n\_1, instructions\_1)}}
      \rule{0.5\textwidth}{0.4pt} 
      \centerline{\parbox{0.5\textwidth}{\tt if condition\_1 then instruction\_1 instructions\_2  \\ elif condition\_2 then instructions\_3 \\ … \\ else instructions\_4 endif}}
      ~\\{$\longrightarrow$}\\~
      \centerline{\parbox{0.5\textwidth}{\tt if condition\_1 then instruction\_1 endif \\  if condition\_1 then instructions\_2  endif \\  if not condition\_1 and condition\_2 then instructions\_3 endif \\  ...\\  if not condition\_1 and not condition\_2 then instructions\_4 endif}}
      \rule{0.5\textwidth}{0.4pt}

      ~\\{$\longrightarrow$}\\~

      \rule{0.5\textwidth}{0.4pt} 
  }\end{tabular}}\begin{itemize}
   \item Introducing $\log(\cdot)$ and $\exp(\cdot)$ makes products, thus polynomials, implementable.
   \item Introducing polynomial, makes regular functions approximatable.

\begin{frame}
  \frametitle{The notion of ``neuronoid'':}
This names the not very biologically plausible simplest biological neuron inspired by mean-field modeling of the Hodgkin–Huxley neuronal axon model:
\eqline{\tau \frac{\partial v_i}{\partial t}(t) + v_i(t) = z_i(t),  z_i(t) \defq \mbox{h}\left(\sum_j \, w_{ij} \, v_j(t) + w_{i0}\right),}
using the normalized sigmoid with: \eqline{\mbox{h}(-\infty) = 0, \mbox{h}(0) = 1/2, \mbox{h}'(0) = 1, \mbox{h}(+\infty) = 1, \mbox{h}(x) = 1 - \mbox{h}(-x).}
~\\{\small \begin{itemize}
   \item A neuronoid $z \rightarrow \left. h_{\tau = 0}(\omega \, z)\right|_{\omega > 10}$ approximates $H(\cdot)$ below $10^{-10}$. 
   \item A neuronoid $z \rightarrow \left. \omega \, h_{\tau = 0}(\frac{z}{\omega})\right|_{\omega > 100}$ approximates $z \rightarrow z$ below $10^{-3}$. 
\end{itemize}} \centerline{
\hphantom{5cm}\includegraphics[width=0.25\textwidth]{./mollification.jpg} 
~~~~~~~~~~~~~~\includegraphics[width=0.25\textwidth]{./linearapproximation.jpg}
}\end{frame}

\begin{frame}[shrink=30] 
  \frametitle{Implemented functions 1/3}
  For binary variables $\mbox{{\tt b\_$\bullet$}} \in \{0, 1\}$, continuous variables $\mbox{{\tt v\_$\bullet$}} \in [-1, 1]$, and the sigmoid $h(\cdot)$ the following
functions, defined previously, are implemented: \\ \centerline{\small \begin{tabular}{ll}&\\
  {\tt v\_o = h(v)} & The normalized sigmoid. \\
  {\tt v\_o = H(v)} & Generalized step-function. \\& $\mbox{{\tt v\_o}} \in \{0_{v<0}, 1/2_{v=0}, 1_{v>0}\}$ \\& Converted to {\tt h()}  when the option {\tt prgm["neuronoid"] =  true}.\\
  {\tt v\_o = Id(v)} & Identity function\\& Converted to a neuronoid when the option {\tt prgm["neuronoid"] =  true}.\\
  {\tt b\_o = And(b\_1, …)} & Conjunction with a variable number of binary  arguments.\\& Implements {\tt b\_1 and b\_2 …} \\
  {\tt b\_o = Or(b\_1, …)} & Disjunction with a variable number of binary arguments.\\& Implements {\tt b\_1 or b\_2 …} \\
  {\tt b\_o = Not(b)} & Negation of a binary value. \\& Implements {\tt not b} \\
  {\tt b\_ o = If\_b(b\_1, b'\_1, …, b'\_0)} & Binary conditional expression, \\& Implements {\tt if b\_1 then b'\_1 …}. \\
  {\tt v\_o = If\_v(b\_1, v\_1, …, v\_0)} & Valued conditional expression, \\& Implements {\tt if b\_1 then v\_1 …}. \\
  {\tt v\_o = Exp\_v(v\_1)} & Exponential function approximation \\& Implements $v_1 \in [-1, 1], e^{v_1} \in [1/e, e] \simeq [0.368, 2.718]$ \\
  {\tt v\_o = Log\_v(v\_1)} & Natural logarithm approximation \\& Implements $v_1 \in [1, e], \log(v_1) \in [0, 1]$ \\
  {\tt v\_o = Prod\_v(v\_1, v\_2)} & Multiplication approximation \\& Implements $v_i \in [-1, 1], i \in \{1, 2\}, v_1 \, v_2 \in  [-1, 1]$ \\
  {\tt v\_o = Prod\_b(b\_1, v\_1, …)} & Sum of binary weights values\\& Implements {\tt b\_1 \, v\_1 + …}.\\& Default missing value is $0$.\\
\end{tabular}} 
\end{frame}
\begin{frame}[shrink=30]

  {\tt Latch\_v(b\_o, v\_i, b\_c)} & Valued memory latch. \\& {\tt b\_o} is the  output, \\& {\tt v\_i} is the input,  \\& $\mbox{{\tt b\_c}} \in \{0_{open}, 1_{latched}\}$ is the control. \\
  {\tt Bistable(b\_o, b\_1, b\_0)} & Two inputs bi-stable latch. \\& {\tt b\_o} is the  output, \\& {\tt b\_0} resets to $0$, \\& {\tt b\_1} sets to $1$. \\
  {\tt Bistable(b\_o, b\_i)} & One input bi-stable latch. \\& {\tt b\_o} is the  output, \\& {\tt b\_i} is the two-way input.\\
  {\tt Spikeup(b\_o, b\_i)} & Spike generation on input rising front. \\& {\tt b\_i} is the input, which rising front is detected.\\
