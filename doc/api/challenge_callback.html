 33def next_output_from_network(context):
 34    """Returns the next bot output and update the state.
 35
 36    Parameter
 37    ----------
 38    context: dictionary
 39    - "model" : The model parameters list as defined below.
 40        - It is a `W_in, W, W_out, warmup, leak, f, g` list for this next_output_from_network() callback, where:
 41            - `p = bot.camera.resolution`.
 42            - `n` # The network size
 43            - `W_in(n,2*p+3)` is the input weights.
 44            - `W(n,n)` is the internal weights, thus `W(row-count,column-count)`.
 45            - `W_out(1,n)` is the output weights.
 46            - `warmup` is the number of iteration, before the bot starts.
 47            - `leak(n, 1)`  the leak vector, or scalar.
 48            - `f = tanh`  is the internal non-linearity.
 49            - `g = identity`  is the internal non-linearity.
 50    - "state"   : The model current input and state.
 51        - It is a `I, X` list where:
 52            - `I[:p] = 1 - bot.camera.depths`.
 53            - `I[p:2*p] = bot.camera.values`.
 54            - `I[2*p,2*p+3] = (bot.hit, bot.energy, 1.0)`.
 55            - `X[0:network_size(),0]` is the local state.
 56    """
 57    W_in, W, W_out, warmup, leak, f, g = context["model"]
 58    I, X = context["state"]
 59    X = leak*X + (1-leak)*f(np.dot(W_in, I) + np.dot(W, X))
 60    O = np.dot(W_out, g(X))
 61    context["state"] = I, X
 62    return O

 64def default_model(n):
 65    bot = Bot()
 66    p = bot.camera.resolution
 67    np.random.seed(1234567)
 68    Win  = np.random.uniform(-1,1, (n,2*p+3))
 69    W = np.random.uniform(-1,1, (n,n))*(np.random.uniform(0,1, (n,n)) < 0.1)
 70    Wout = 0.1*np.random.uniform(-1, 1, (1,n))
 71    warmup = 0
 72    f = sigmoid
 73    g = identity
 74    leak = 0
 75    model = Win, W, Wout, warmup, leak, f, g
 76    return model

116        import matplotlib.pyplot as plt

122        world = environment.world

128        ax2 = plt.axes([1/2,0.0,1/2,1.0], aspect=1, frameon=False)

135            "bot" : ax1.add_artist(Circle((0,0), 0.05,
136                                          zorder=50, facecolor="white", edgecolor="black")),
137            "rays" : ax1.add_collection(LineCollection([], color="C1", linewidth=0.5, zorder=30)),
138            "hits" :  ax1.scatter([], [], s=1, linewidth=0, color="black", zorder=40),
139            "camera" : ax2.imshow(np.zeros((1,1,3)), interpolation="nearest",
140                                  origin="lower", extent = [0.0, 1.0, 0.0, 1.0]),
141            "energy" : ax2.add_collection(
142                LineCollection([[(0.1, 0.1),(0.9, 0.1)],
143                                [(0.1, 0.1),(0.9, 0.1)],
144                                [(0.1, 0.1),(0.9, 0.1)]],

157

160        context = dict({"model": model, "state": (I, X)})
161
162        distance = 0
163        hits = 0
164        iteration = 0
165
166        # Initial update
167        if debug:
168            bot.camera.render(bot.position, bot.direction,
169                              environment.world, environment.colormap)
170        else:
171            bot.camera.update(bot.position, bot.direction,
172                              environment.world, environment.colormap)
173
174        # Run until no energy
175        while bot.energy > 0:
176
177            energy = bot.energy
178
179            p = bot.camera.resolution
180            I[:p,0] = 1 - bot.camera.depths
181            I[p:2*p,0] = bot.camera.values

183

186            # During warmup, bot does not move

210                                                     [(0.1, 0.1),(0.9, 0.1)]])            
211                graphics["camera"].set_data(bot.camera.framebuffer)
212                bot.camera.render(bot.position, bot.direction,
213                              environment.world, environment.colormap)
214                plt.pause(1/60)
215
216        scores.append(distance)
217    if debug:
218        elapsed = time.time() - start_time
219        print(f"Evaluation completed after {elapsed:.2f} seconds")
220        print(f"Final score: {np.mean(scores):.2f} ± {np.std(scores):.2f}")
221
222    return np.mean(scores), np.std(scores)
